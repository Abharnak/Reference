Name Display =
VAR CurrentName = SELECTEDVALUE(Table[name])
VAR PrevName =
    CALCULATE(
        MAX(Table[name]),
        FILTER(
            ALLSELECTED(Table),
            Table[Index] = MAX(Table[Index]) - 1
        )
    )
RETURN
IF(CurrentName = PrevName, BLANK(), CurrentName)
