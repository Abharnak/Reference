VAR AllAvailable =
    CALCULATE(
        MIN(Table[Availability]),
        ALLEXCEPT(Table, Table[Customer])
    ) = "available"

VAR AnyUnavailable =
    CALCULATE(
        MAX(Table[Availability]),
        ALLEXCEPT(Table, Table[Customer])
    ) <> "available"

VAR TotalSales =
    CALCULATE(
        SUM(Table[Sales]),
        ALLEXCEPT(Table, Table[Customer])
    )

RETURN
IF(
    AllAvailable,
    TotalSales,
